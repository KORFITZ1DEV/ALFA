#Rect also moving in y-direction

int len = 20;
rect myRect1 = createRect(0,0,len,len);

int offsetX = 100;
int offsetY = 100;
int duration = 1500;

loop(int i from 1 .. 4) {
    move(myRect1, offsetX, 0, duration);
    move(myRect1, 0, offsetY, duration);
    move(myRect1, -offsetX, 0, duration);
    move(myRect1, 0, -offsetY, duration);
}
